def to_lock
  out = "  #{name}"
  unless requirement == Gem::Requirement.default
    reqs = requirement.requirements.map{|o,v| "#{o} #{v}" }.sort.reverse
    out << " (#{reqs.join(', ')})"
  end
  out
end

def to_yaml_properties

end
