def options=(options)
  @options = options
  children.each do |c|
    if c.is_a? Hash
      c.values.each {|v| v.options = options}
    else
      c.options = options
    end
  end
end

def dasherize(s, opts)
  if opts[:dasherize]
    s.gsub(/_/, '-')
  else
    s
  end
end
